	int Maxchislo(int nomer) {
		int Max = 0;
		while (nomer > 0) {
			int cifra = nomer % 10;
			if (cifra > Max) {
				Max = cifra;
			}
			nomer = nomer / 10;
		}
		return Max;
	}
	int Minchislo(int nomer) {
		int Min = 9;
		while (nomer > 0) {

			int cifra = nomer % 10; 
			if (cifra < Min) {
				Min = cifra;
			}
			nomer = nomer / 10;
		}
		return Min;
	}
